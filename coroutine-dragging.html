This demo of a coroutine event loop uses features of Javascript 1.7 and thus only works in Firefox.

Try dragging the box around, and observe how the code works:

var loop = coroutine(function() {
    var event;
    while (event = yield) { // not dragging
        if (event.type == 'mousedown') {
            while (event = yield) { // dragging
                if (event.type == 'mousemove')
                    move(event);
                if (event.type == 'mouseup')
                    break;
            }
        }
        // ignore mousemoves
    }
});

$('#box').mousedown(loop);
$(window).mousemove(loop)
    .mouseup(loop);
